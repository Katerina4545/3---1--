CREATE TABLE Boat(
id_boat NUMBER(3) NOT NULL CONSTRAINT BOAT_PK_ID PRIMARY KEY
name VARCHAR2(15) NOT NULL
port_write VARCHAR2(15) NOT NULL
exemption NUMBER(3, 2)


CONSTRAINT BOAT_FK_PORT 
FOREIGN KEY (port_write) REFERENCES Place_Put(port)
)

CREATE TABLE Cargo(
id_cargo NUMBER(3) NOT NULL CONSTRAINT CARGO_PK_ID PRIMARY KEY
name VARCHAR2(15) NOT NULL
port_put VARCHAR2(15) NOT NULL
max_count NUMBER(50, 2) CONSTRAINT CARGO_CH_COUNT CHECK( max_count > 0)
cost NUMBER(50, 2) CONSTRAINT CARGO_CH_COST CHECK( cost > 0 )


CONSTRAINT CARGO_FK_PORT 
FOREIGN KEY (port_write) REFERENCES Place_Put(port) 
)

CREATE TABLE Place_Put(
id_place_put NUMBER(3) NOT NULL CONSTRAINT PLACE_PUT_PK_ID PRIMARY KEY
berth VARCHAR2(15) NOT NULL
port VARCHAR2(15) NOT NULL
allocation NUMBER(3, 2)
)

CREATE TABLE Loading(
number_statement NUMBER(3) NOT NULL CONSTRAINT LOADING_PK_NUMBER PRIMARY KEY
date DATE
boat NUMBER(3) NOT NULL 
place_put NUMBER(3) NOT NULL
cargo NUMBER(3) NOT NULL
count NUMBER(10) NOT NULL
cost NUMBER(10) 

CONSTRAINT LOADING_FK_CARGO 
FOREIGN KEY (cargo) REFERENCES Cargo(id_cargo),


CONSTRAINT LOADING_FK_PLACE_PUT 
FOREIGN KEY (place_put) REFERENCES Place_Put(id_place_put),


CONSTRAINT LOADING_FK_BOAT 
FOREIGN KEY (boat) REFERENCES Boat(id_boat)
)
